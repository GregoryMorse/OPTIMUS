108void nelderMead::step()
109{
110
111start:
112 order();
113 center();
114 ybestPoint = population.begin()->first;
115 worst = (population.end())->first;
116 secondWorst = population[population_count - 2].first;
117 reflection();
118 if (yreflectedPoint < secondWorst && yreflectedPoint > ybestPoint)
119 {
120 population[population_count - 1].second = reflectedPoint;
121 population[population_count - 1].first = yreflectedPoint;
122 goto start;
123 }
124 else if (yreflectedPoint < ybestPoint)
125 {
126 expansion();
127 if (yexpandedPoint < yreflectedPoint)
128 {
129 population[population_count - 1].second = expandedPoint;
130 population[population_count - 1].first = yexpandedPoint;
131 goto start;
132 }
133 else
134 {
135 population[population_count - 1].second = reflectedPoint;
136 population[population_count - 1].first = yreflectedPoint;
137 goto start;
138 }
139 goto start;
140 }
141 else
142 {
143 if (yreflectedPoint < worst)
144 {
145 contraction();
146 if (ycontractedPoint < yreflectedPoint)
147 {
148 population[population_count - 1].second = contractedPoint;
149 population[population_count - 1].first = ycontractedPoint;
150 goto start;
151 }
152 else
153 shrink();
154 }
155 else
156 {
157 if (ycontractedPoint < worst)
158 {
159 contractionB();
160 population[population_count - 1].second = contractedPoint;
161 population[population_count - 1].first = ycontractedPoint;
162 goto start;
163 }
164 else
165 shrink();
166 }
167 }
168 newSum = accumulate(&population.begin()->first, &population.end()->first, 0);
169 newSum = newSum / population_count;
170 ++generation;
171}
