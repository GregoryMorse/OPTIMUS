124 {

128 for(int j=0;j<myProblem->getDimension();j++)
129 {
130 double rt1 = myProblem->randomDouble();
131 double rt2 = myProblem->randomDouble();
132 velocity[i][j]=velocity[i][j]+rt1 * r1 * (bestPosition[i][j]-position[i][j])+
133 rt2 * r2 * (bestg[j]-position[i][j]);
134 double tryValue = position[i][j]+velocity[i][j];
135 if(tryValue>=lmargin[j] && tryValue<=rmargin[j])
136 position[i][j]=tryValue;
