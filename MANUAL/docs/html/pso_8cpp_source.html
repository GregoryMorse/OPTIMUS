39bool Pso::terminated()
40{
41 int max_generations=params["pso_generations"].toString().toInt();
42 double fmin=fabs(1.0+fabs(besty));
43 if(generation<=1) {x1=0.0;x2=0.0;}
44 x1+=fmin;
45 x2+=fmin * fmin;
46 variance = x2/(generation+1) -(x1/(generation+1))*(x1/(generation+1));
47 variance=fabs(variance);
48 if(besty<oldbesty)
49 {
50 oldbesty=besty;
51 stopat=variance/2.0;
52 }
53 if(stopat<1e-8 && !isnan(variance)) stopat=variance/2.0;
54 printf("Generation %5d value: %12.8lg variance: %lf stopat: %lf\n",generation,besty,
55 variance,stopat);
56 return generation>=max_generations|| (variance<=stopat && generation>=10);
57}

118void Pso::calcFitnessArray()
119{
120 int genome_count=params["pso_particles"].toString().toInt();
121 double wmin=params["pso_inertia_start"].toString().toDouble();
122 double wmax=params["pso_inertia_end"].toString().toDouble();
123 int maxGenerations=params["pso_generations"].toString().toInt();
124 Data distances;
125 int genome_size=myProblem->getDimension();
126 double inertia= wmax-generation*1.0/maxGenerations*(wmax-wmin);
127 double c1=params["pso_c1"].toString().toDouble();
128 double c2=params["pso_c2"].toString().toDouble();
129
130 double localsearch_rate=params["pso_localsearch_rate"].toString().toDouble();
131 for(int i=0;i<genome_count;i++)
132 {
133 Data oldg=particle[i];
134 for(int j=0;j<genome_size;j++)
135 {
136 double r1=myProblem->randomDouble();
137 double r2=myProblem->randomDouble();
138
139 double tj=velocity[i][j];
140 double part1=inertia * velocity[i][j];
141 double part2=c1 * r1 * (bestParticle[i][j]-particle[i][j]);
142 double part3=c2 * r2 * (bestx[j]-particle[i][j]);
143 velocity[i][j]=part1+part2+part3;
144 double trialf=particle[i][j]+velocity[i][j];
145 if(trialf<lmargin[j] || trialf>rmargin[j])
146 {
147 velocity[i][j]=tj;
148 continue;
149 }
150 else
151 {
152 particle[i][j]=trialf;
153
154 }
155 }
156 distances.push_back(getDistance(particle[i],oldg));
157 }
158#pragma omp parallel for num_threads(threads)
159 for(int i=0;i<genome_count;i++)
160 {
161 if(distances[i]>1e-6 )
162 {
163 if(localsearch_rate>0.0 && myProblem->randomDouble()<=localsearch_rate
164 && !checkGradientCriterion(particle[i]))
165 {
166 Data dg=particle[i];
167 fitness_array[i]=localSearch(particle[i]);
168 RC+=getDistance(dg,particle[i]);
169 localSearchCount++;
170 #pragma omp critical
171 {
172 bool found=false;
173 for(int j=0;j<minimax.size();j++)
174 {
175 if(getDistance(minimax[j],particle[i])<1e-5) {found=true;break;}
176 }
177 if(!found)
178 {
179 minimax.push_back(particle[i]);
180 }
181 minimax.push_back(dg);
182 }
183 }
184 else
185 fitness_array[i]=fitness(particle[i]);
186 }
187 }
