288 {
289  int start = 1;

291  const double b=(1.0-generation*1.0/maxGenerations)*5.0;
292  double mutation_rate=params["mutation_rate"].toString().toDouble();
293  for(int i=start;i<chromosome_count;i++)
294  {
295  for(int j=0;j<chromosome_size;j++)
296  {
297  double r=myProblem->randomDouble();
298  if(r<mutation_rate)
299  {
300  double old=chromosome[i][j];
301 #ifdef MPT_MUTATION
302  double t,t2,rk,beta=5.0;
303  t=(genome[i][j]-lmargin[j])/(rmargin[j]-genome[i][j]);
304  do
305  {
306  rk=problem->randomDouble();
307  if(rk<t) t2=t-t*pow((t-rk)/t,beta);
308  else
309  if(rk==t) t2=t;
310  else
311  t2=t+(1.0-t)*pow((rk-t)/(1.0-t),beta);
312  genome[i][j]=(1.0-t2)*lmargin[j]+t2*rmargin[j];
313  }while(genome[i][j]<lmargin[j] || genome[i][j]>rmargin[j]);
314 #endif
315 #ifdef PSO_MUTATION
316  do{
317  genome[i][j]=problem->randomDouble()*(bestgenome[i][j]-genome[i][j])+drand48()*(genome[0][j]-genome[i][j]);
318  }while(genome[i][j]<lmargin[j]||genome[i][j]>rmargin[j]);
319 #endif
320 #ifdef RANDOM_MUTATION
321  do{
322  genome[i][j]*=2.0*problem->randomDouble()-1.0;
323  }while(genome[i][j]<lmargin[j]||genome[i][j]>rmargin[j]);
324 #endif
325 #ifdef DOUBLE_MUTATION
326  double xi,yi;
327  do{
328  xi=myProblem->randomDouble();
329  yi=myProblem->randomDouble();
330  if(xi>0.5)
331  {
332  chromosome[i][j]=chromosome[i][j]+(rmargin[j]-chromosome[i][j])*(1.0-pow(yi,b));
333  }
334  else
335  {
336  chromosome[i][j]=chromosome[i][j]-(chromosome[i][j]-lmargin[j])*(1.0-pow(yi,b));
337  }
338  }while(chromosome[i][j]<lmargin[j]
339  || chromosome[i][j]>rmargin[j]);
340 #endif
